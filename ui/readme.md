## vue install 
```bash
  # update npm version to @7
  npm install -g npm@7
  # install vue/cli@3  
  npm install -g @vue/cli@3
  vue --version
  # install vue/cli-init
  npm install -g @vue/cli-init
  vue init --version
  # vue initializer
  vue init nuxt-community/typescript-template ui
```
